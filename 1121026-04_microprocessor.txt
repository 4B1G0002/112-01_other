#include <Adafruit_LEDBackpack.h>
Adafruit_7segment led_display1 = Adafruit_7segment();

// C++ code
//
void setup()
{
  led_display1.begin();
  pinMode(A1, INPUT);
}

void loop()
{
  for (int i = 0; i <= 1200; i++){
    led_display1.println(i);
    led_display1.writeDisplay();
    delay(100);
  }
}
